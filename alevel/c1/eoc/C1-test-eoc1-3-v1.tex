    \[y = x^2 + ax + b\]

%%% Question %%%
\question
\begin{parts}
\part[4]
	By completing the square, find in terms of the constant $k$ the roots of the equation
    \[x^2 + 4kx - k=0\].
\part[4]
	Hence find the set of values of $k$ for which the equation has no real roots.
\end{parts}
\sol{\fill}

%%% Question %%%
\question The line $l$ has the equation $y=2x-3$ and the curve $C$ has equation $y=(x+2)(x-4)$.
\begin{parts}
\part[5] 
	Sketch the line $l$ and the curve $C$ on the same axes, showing the coordinates of the $x$- and $y$- intercepts.
\part[2]
	Show that the $x$-coordinates of the points of intersection of $l$ and $C$ satisfy the equation $x^2-4x-5=0$.
\part[4]
	Hence, or otherwise, find the points of intersection of $l$ and $C$.
\end{parts}
\vfill
\sol{5 in}
